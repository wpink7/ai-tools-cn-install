# 使用 curl
curl -o- https://gitee.com/mirrors/nvm/raw/master/install.sh | bash

# 或者使用 wget (和你的原始命令风格一致)
wget -qO- https://gitee.com/mirrors/nvm/raw/master/install.sh | bash

# 1. 克隆仓库到本地
git clone https://gitee.com/mirrors/nvm.git ~/.nvm
# 2. 进入目录并执行安装脚本
cd ~/.nvm
./install.sh
# 3. 手动配置环境变量 (将以下内容添加到 ~/.bashrc 或 ~/.zshrc 文件中)
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # 加载 nvm
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion" # 加载 nvm 自动补全
# 4. 使配置生效
source ~/.bashrc

npm install -g @anthropic-ai/claude-code

打开命令提示符 (CMD) 或 PowerShell，运行：

安装 OpenClaw
可通过下面的 NPM 命令安装 OpenClaw 官方 CLI 程序
npm install -g openclaw@latest

配置 OpenClaw
openclaw onboard --install-daemon
openclaw config
